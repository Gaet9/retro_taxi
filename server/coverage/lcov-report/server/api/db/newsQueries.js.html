const create_newsletter = async (req, res) => {
    try {
        // Step 1: Call Perplexity API
        const perplexityRes = await fetch("https://api.perplexity.ai/chat/completions", {
            method: "POST",
            headers: {
                Authorization: `Bearer ${perplexityKey}`, // Store this in your .env or config.js
                "Content-Type": "application/json",
            },
            body: JSON.stringify({
                model: "sonar", // or another supported model
                messages: [
                    {
                        role: "user",
                        content: `Give me a summary of the latest global news about autonomous vehicles around the world. Include key developments, companies, and regulatory updates.`,
                    },
                ],
            }),
        });

        if (!perplexityRes.ok) {
            const errorText = await perplexityRes.text();
            console.error("Perplexity API error:", errorText);
            throw new Error("Failed to fetch from Perplexity API");
        }

        const perplexityData = await perplexityRes.json();
        const newsSummary = perplexityData.choices[0].message.content;

        // Step 2: Build prompt for GPT
        const prompt = `
                        You are an AI newsletter journalist covering global autonomous vehicle developments.

                        Use the following article summaries to generate a JSON newsletter with this structure:

                        [
                        {
                            "title": "Retro taxi – Week of ${dayjs().format("MMMM D")}",
                            "summary": "Maximum 30 words to summarize the content"
                            "category": "One word summarizing the type of news"
                            "model": "to fill only if the news concerns a specific model of a brand"
                            "brand":  "to fill only if the news concerns a specific brand"
                            "content": {
                            "Regulation Updates": ["• point", "• another"],
                            "Industry Highlights": [...],
                            "Market Insights": [...],
                            "Global Hotspots": [...]
                            }
                        }
                        ]

                        Use only this information:
                        ${newsSummary}

                        Return ONLY the JSON structure above.
                        `;

        // Step 3: Call OpenAI GPT to format the newsletter
        const completion = await openai.chat.completions.create({
            model: "gpt-3.5-turbo", // or "gpt-3.5-turbo" cheaper
            messages: [{ role: "user", content: prompt }],
            temperature: 0.7,
        });

        const content = completion.choices[0].message.content;

        // CLEAN the output (Perplexity or GPT might wrap JSON in code blocks)
        let cleaned;
        try {
            cleaned = JSON.parse(content); // try direct parse
        } catch (err) {
            // Fallback: try cleaning the markdown wrapping
            const fallback = content.replace(/^```json\s*/i, "").replace(/```$/, "");
            cleaned = JSON.parse(fallback);
        }

        const newsletter = Array.isArray(cleaned) ? cleaned[0] : cleaned;

        newsletter.user_id = 14; // or from req.user.id or similar

        await pool.query(
            `INSERT INTO blogs (title, summary, content, category, model, brand, user_id) VALUES ($1, $2, $3, $4, $5, $6, $7)`,
            [
                newsletter.title,
                newsletter.summary,
                JSON.stringify(newsletter.content),
                newsletter.category,
                newsletter.model,
                newsletter.brand,
                newsletter.user_id,
            ]
        );

        // Step 4: Parse and return JSON
        res.json(cleaned);
    } catch (err) {
        console.error("Error generating newsletter:", err.message);
        res.status(500).json({ error: "Something went wrong" });
    }
};
 
const get_newsletters = async (req, res) => {
    try {
        const result = await pool.query(`SELECT * FROM newsletters ORDER BY created_at DESC`);
        const newsletter = result.rows[0];

        if (!newsletter) {
            return res.status(404).json({ success: false, error: "No newsletter found" });
        }

        res.json(newsletter);
    } catch (err) {
        console.error("Error fetching newsletter:", err.message);
        res.status(500).json({ success: false, error: "Server error" });
    }
};

const send_newsletter = async (req, res) => {
    const resend = new Resend(config.apiKeys.resend);

    try {
        const result = await pool.query(`SELECT * FROM newsletters ORDER BY created_at DESC LIMIT 1`);
        const newsletter = result.rows[0];
        const content = JSON.parse(newsletter.content);

        // Generate basic HTML from content
        const html = `
            <h1>'Hi there '${req.body.name}'!'</h1>
            <h1>${newsletter.title}</h1>
            <p><em>${newsletter.created_at}</em></p>
            <p>${newsletter.summary}</p>
            <hr />
            ${Object.entries(content)
                .map(
                    ([content, points]) => `
                <h2>${content}</h2>
                <ul>${points.map((p) => `<li>${p}</li>`).join("")}</ul>
            `
                )
                .join("<br/>")}
        `;

        const response = await resend.emails.send({
            from: "RetroTaxi 🚖 <hello@retrotaxi.xyz>",
            to: req.body.email,
            subject: newsletter.title,
            html,
            reply_to: "support@retrotaxi.xyz",
        });
        console.log("Email sent:", response);
        res.status(200).json({ message: `Newsletter sent to ${req.body.email}`, id: response.id });
    } catch (err) {
        console.error("Failed to send newsletter:", err.message);
        res.status(500).json({ error: "Failed to send newsletter" });
    }
};
